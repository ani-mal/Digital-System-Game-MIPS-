

	LOADCTE reg11 65 ; CODIGO DA TECLA a	
	LOADCTE reg10 68 ; CODIGO DA TECLA d
	LOADCTE reg9 87 ; CODIGO DA TECLA w
	LOADCTE reg8 83 ; CODIGO DA TECLA s
	LOADCTE reg6 88 ; CODIGO DO CARACTER X
	LOADCTE reg5 20 ; PASSO PARA SUBIR OU DESCER NA TELA

	LOADCTE reg0 00  ; Limpa a Tela
	MOV reg1 reg0 ; ZERA REG1
	LOADCTE reg15 01  ; CONSTANTE = 01
	LOADCTE reg14 239  ; CONSTANTE = EF -> FIM DA TELA
	LOADCTE reg13 255  ; CONSTANTE = FF

LOCO:	OUTCHAR reg0 reg1 ;ZERA A TELA
	ADD reg1 reg15   ; 
	COMP reg1 reg14
	JMPE VAIVAI
		
	JMP LOCO

VAIVAI:	LOADCTE reg0 00  ; 
LALA:	NOP
VVVV:	INCHAR reg1; LE TECLADO
	COMP reg1 reg13 ; VERIFICA SE E DIFERENTE DE 'FF'
	JMPE VVVV
	
	OUTCHAR reg1 reg0 ; ESCREVE TECLA PRESSIONADA

	OUTCHAR reg0 reg7 ; APAGA X'zinho

	COMP reg1 reg11		; VERIFICA SE A TECLA A FOI PRESSIONADA
	JMPE AA

	COMP reg1 reg10		; VERIFICA SE A TECLA D FOI PRESSIONADA
	JMPE DD

	COMP reg1 reg9		; VERIFICA SE A TECLA W FOI PRESSIONADA
	JMPE WW

	COMP reg1 reg8		; VERIFICA SE A TECLA S FOI PRESSIONADA
	JMPE SS

CONT:	OUTCHAR reg6 reg7; ESCREVE X'zinho

	LOADCTE reg12 25000  ; Delay Loop
DELAY:	SUB reg12 reg15
	COMP reg12 reg0
	JMPE LALO
	JMP DELAY

LALO:	NOP
	JMP LALA

AA:	SUB reg7 reg15 ; REG7 <- REG7 - 1
	JMP CONT

DD:	ADD reg7 reg15 ; REG7 <- REG7 + 1
	JMP CONT

WW:	SUB reg7 reg5 ; REG7 <- REG7 - 20
	JMP CONT

SS:	ADD reg7 reg5 ; REG7 <- REG7 +20
	JMP CONT
